# Convert Date
df['Date'] = pd.to_datetime(df['Date'])

# Extract features
df['Month'] = df['Date'].dt.month
df['Hour'] = df['Date'].dt.hour

# Email length
df['Email_Length'] = df['Body'].apply(len)
